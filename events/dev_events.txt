namespace = dev_events

country_event = {
	# adm_click
	id = dev_events.3
	title = dev_events.3.t
	desc = dev_events.3.d
	picture = INVENTION_eventPicture
	hidden = yes
	
	trigger = {
		OR = {
			AND = {
				ai = yes
				has_global_flag = ai_dev_click
			}
			AND = {
				NOT = { ai = yes }
				has_country_flag = dev_click
			}
		}
		
		OR = {
			had_country_flag = { flag = admClick	days = 31 }
			NOT = { has_country_flag = admClick }
		}
		
		power_tech = {
			power = adm
			value = 960
		}
	}
	
	immediate = {
		hidden_effect = {
			clr_country_flag = admClick
			set_country_flag = admClick
			
			set_base_target = {
				which = base_target
				base = adm
			}
			if = {
				limit = {
					OR = {
						is_migratory_tribe = yes
						check_variable = {
							which = baseCost
							value = 999
						}
					}
				}
				if = {
					limit = {
						NOT = { stability = 3 }
					}
					stab_cost = yes
					add_stability = 1
					log = "[This.GetName] raises stability**"
				}
				else = {
					adm_power_cost = 75
					add_inflation = -2
					log = "[This.GetName] reduces inflation**"
				}
			}
			else = {
				event_target:base_target = {
					add_power0 = {
						power = adm
					}
					add_next_institution_embracement_scaled = 5
					add_base_tax = 1
					add_devastation_reduction = yes
					log = dev_info.1
				}
				change_estate_land_share = {
					estate = all
					share = -0.2
				}
				change_variable = {
					which = admClicks
					value = 1
				}
				change_variable = {
					which = devsumClicks
					value = 1
				}
			}
		}
	}
	
	option = {
		#hidden
	}
}
