MIN_value = {
	if = {
		limit = {
			check_variable = {
				which = $which$
				value = $value$
			}
		}
		set_variable = {
			which = $which$
			value = $value$
		}
	}
}

MAX_value = {
	if = {
		limit = {
			NOT = {
				check_variable = {
					which = $which$
					value = $value$
				}
			}
		}
		set_variable = {
			which = $which$
			value = $value$
		}
	}
}

set_base_value = {
	set_variable = { #100% base cost
		which = $which$
		value = 1 
	}
	
	export_to_variable = {
		variable_name = apowerCost
		value = modifier:all_power_cost
		who = owner
	}
	change_variable = { #power cost modifiers
		which = $which$
		which = apowerCost
	}
	
	export_to_variable = {
		variable_name = devCost
		value = modifier:development_cost
		who = owner
	}
	change_variable = { #country dev modifiers
		which = $which$
		which = devCost
	}
	
	export_to_variable = {
		variable_name = ldevCost
		value = modifier:local_development_cost
	}
	change_variable = { #province dev modifiers
		which = $which$
		which = ldevCost
	}
	
	#same day development
	export_to_variable = {
		variable_name = pDev
		value = trigger_value:development
	}
	set_variable = {
		which = counter
		which = devCount
	}
	while = {
		limit = {
			check_variable = {
				which = counter
				value = 1
			}
		}
		subtract_variable = {
			which = counter
			value = 1
		}
		set_variable = {
			which = targetDev
			which = pDev
		}
		subtract_variable = {
			which = targetDev
			which = counter
		}
		
		if = {
			limit = {
				check_variable = {
					which = targetDev
					value = 40
				}
			}
			change_variable = {
				which = $which$
				value = 0.12
			}
		}
		else_if = {
			limit = {
				check_variable = {
					which = targetDev
					value = 30
				}
			}
			change_variable = {
				which = $which$
				value = 0.09
			}
		}
		else_if = {
			limit = {
				check_variable = {
					which = targetDev
					value = 20
				}
			}
			change_variable = {
				which = $which$
				value = 0.06
			}
		}
		else_if = {
			limit = {
				check_variable = {
					which = targetDev
					value = 10
				}
			}
			change_variable = {
				which = $which$
				value = 0.03
			}
		}
	}
	
	if = { #5% in capital per 100 country dev (up to 50%)
		limit = {
			is_capital = yes
		}
		export_to_variable = {
			variable_name = totalDev
			value = total_development
			who = owner
		}
		if = { 
			limit = {
				check_variable = {
					which = totalDev
					value = 1000
				}
			}
			set_variable = {
				which = totalDev
				value = -0.5
			}
		}
		else = {
			divide_variable = {
				which = totalDev
				value = -2000
			}
		}
		change_variable = { 
			which = $which$
			which = totalDev
		}
	}
	
	if = { #dev efficiency from adm tech affects base cost by -10%
		limit = {
			owner = { adm_tech = 27 } 
		}
		multiply_variable = { 
			which = $which$
			value = 35
		}
	}
	else_if  = {
		limit = {
			owner = { adm_tech = 23 } 
		}
		multiply_variable = { 
			which = $which$
			value = 40
		}
	}
	else_if  = {
		limit = {
			owner = { adm_tech = 17 }
		}
		multiply_variable = { 
			which = $which$
			value = 45
		}
	}
	else = {
		multiply_variable = { 
			which = $which$
			value = 50
		}
	}
}

set_base_target = {
	set_country_flag = $base$_flag
	
	set_variable = {
		which = baseCost
		value = 999
	}
	every_owned_province = {
		limit = {
			is_state_core = owner
		}
		
		export_to_variable = { which = bTax 	value = base_tax }
		export_to_variable = { which = bPro 	value = base_production }
		export_to_variable = { which = bMan 	value = base_manpower }
		
		set_variable 	= { which = notTax 	which = bPro }
		change_variable = { which = notTax 	which = bMan }
		
		set_variable 	= { which = notPro 	which = bMan }
		change_variable = { which = notPro 	which = bTax }
		
		set_variable 	= { which = notMan 	which = bTax }
		change_variable = { which = notMan 	which = bPro }
		
		
		if = {
			limit = {
				owner = {
					has_country_flag = adm_flag
				}
				NOT = {
					check_variable = {
						which = notTax
						which = bTax
					}
				}
			}
		}
		else_if = {
			limit = {
				owner = {
					has_country_flag = dip_flag
				}
				NOT = {
					check_variable = {
						which = notPro
						which = bPro
					}
				}
			}
		}
		else_if = {
			limit = {
				owner = {
					has_country_flag = mil_flag
				}
				NOT = {
					check_variable = {
						which = notMan
						which = bMan
					}
				}
			}
		}
		else_if = {
			limit = {
				OR = {
					NOT = { controlled_by = owner }
					has_siege = yes
				}
			}
		}
		else = {
			set_base_value = {
				which = baseCost
			}
			
			if = { #dip prefer gold
				limit = {
					owner = {
						has_country_flag = dip_flag
					}
					trade_goods = gold
				}
				owner = {
					multiply_variable = {
						which = baseCost
						value = 3
					}
				}
				
				if = {
					limit = {
						NOT = {
							check_variable = {
								which = baseCost
								which = PREV
							}
						}
					}
					owner = {
						set_variable = {
							which = baseCost
							which = PREV
						}
					}
					save_event_target_as = $which$
				}
				
				owner = {
					divide_variable = {
						which = baseCost
						value = 3
					}
				}
			}
			else_if = { #institutions
				limit = {
					owner = {
						has_country_flag = institution_flag
					}
					OR = {
						$base$ = 100
						NOT = { development = 5 }
						development = 28
						has_terrain = glacier
						has_terrain = desert
						has_climate = arctic
					}
				}
			}
			else_if = {
				limit = {
					NOT = {
						check_variable = {
							which = baseCost
							which = PREV
						}
					}
				}
				owner = {
					set_variable = {
						which = baseCost
						which = PREV
					}
				}
				save_event_target_as = $which$
			}
		}
	}
	
	clr_country_flag = $base$_flag
}

spawn_institution = {
	set_base_target = {
		which = base_target
		base = $institution$
	}
	
	event_target:base_target = {
		set_variable = {
			which = devCount
			value = 0
		}
		
		while = {
			limit = {
				NOT = { $institution$ = 100 }
			}
			set_base_value = {
				which = baseCost
			}
			
			export_to_variable = { variable_name = admPower 	value = trigger_value:adm_power 	who = PREV }
			export_to_variable = { variable_name = dipPower 	value = trigger_value:dip_power 	who = PREV }
			export_to_variable = { variable_name = milPower 	value = trigger_value:mil_power 	who = PREV }
			
			add_next_institution_embracement_scaled = 5			
			
			if = {
				limit = {
					check_variable = { which = admPower 	which = dipPower }
					check_variable = { which = admPower 	which = milPower }
				}
				add_power = {
					power = adm
				}
				add_base_tax = 1
				log = dev_info.1
			}
			else_if = {
				limit = {
					check_variable = { which = dipPower	which = milPower }
				}
				add_power = {
					power = dip
				}
				add_base_production = 1
				log = dev_info.2
			}
			else = {
				add_power = {
					power = mil
				}
				add_base_manpower = 1
				log = dev_info.3
			}
			
			add_devastation_reduction = yes
			owner = {
				change_estate_land_share = {
					estate = all
					share = -0.2
				}
			}
			
			change_variable = {
				which = devCount
				value = 1
			}
		}
		
		set_variable = {
			which = devCount
			value = 0
		}
	}
}
